Código para descobrir raiz quadrada


algoritmo "Descobrindo se um número tem raiz quadrada"
  
var
  encontrouRaiz : logico
  contador: inteiro
  alvo: inteiro

inicio
  encontrouRaiz <- falso
  leia(alvo)
  para i de 1 ate alvo faca
    encontrouRaiz <- i * i = alvo
    contador <- contador + 1
    se encontrouRaiz entao
      escreval("A raiz do ", alvo, " é ", i)
      interrompa
    fimse
  fimpara
  escreval(contador)
  se nao encontrouRaiz entao
    escreval("A raiz do ", alvo, " não é um número inteiro.")
  fimse
fimalgoritmo

Código para fazer sequência Fibonacci

  limite : inteiro
  contador: inteiro
  num1: inteiro
  num2: inteiro
  soma: inteiro

inicio
  leia(limite)
  leia(num1)
  escreva(num1, " ")
  num2 <- num1

  enquanto soma < limite faca
    soma <- num1 + num2
    num1 <- num2
    num2 <- soma
    escreva(soma, " ")
