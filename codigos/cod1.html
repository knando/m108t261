Código contando palavras tirando múltiplos espaços


algoritmo "Primeiro Programa"

var
  palavra: caractere
  letra: caractere
  contador: inteiro
  posicaoOndeEncontreiChar : inteiro


funcao contaPalavras(texto : caractere) : inteiro
  inicio
  contador <- 0

  se Compr(texto) <> 0 entao
    contador <- 1
    enquanto Pos(texto, " ") <> 0 faca
      posicaoOndeEncontreiChar <- Pos(texto, " ")
      se posicaoOndeEncontreiChar <> 0 entao
        contador <- contador + 1
        texto <- Copia(texto, posicaoOndeEncontreiChar + 1, 99)

        enquanto Pos(texto, " ") = 1 faca
          texto <- Copia(texto, 2, 99)
        fimenquanto

      fimse
    fimenquanto
  
  fimse


  retorne contador
fimfuncao

inicio
  palavra <- "Texto  com      espaços    "
  letra <- " "
